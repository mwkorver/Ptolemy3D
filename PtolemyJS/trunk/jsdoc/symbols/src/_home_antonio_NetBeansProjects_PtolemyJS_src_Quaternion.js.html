 26 /**
 27  * Creates a new quaternion instance given four components.
 28  * @constructor
 29  * @param {Number} x
 30  * @param {Number} y
 31  * @param {Number} z
 32  * @param {Number} w
 33  */
 34 Ptolemy.Quaternion = function(x, y, z, w) {
 35     this.x = x;
 36     this.y = y;
 37     this.z = z;
 38     this.w = w;
 39 };
 40 
 41 /**
 42  * Check if this quaternion is equal to the specified one.
 43  * @param {Quaternion} quaternion Quaternion
 44  * @return True if they are equals, false otherwise.
 45  */
 46 Ptolemy.Quaternion.prototype.equals = function(quaternion) {
 47     return (this.x == quaternion.x)
 48     && (this.y == quaternion.y)
 49     && (this.z == quaternion.z)
 50     && (this.w == quaternion.w);
 51 };

 53 /**
 54  * Gets a copy of the Quaternion object as an array. 
 55  * @return Array with a copy of the Quaternion object values.
 56  */
 57 Ptolemy.Quaternion.prototype.toArray = function() {
 58     return [this.x, this.y, this.z, this.w];
 59 };

 61 /**

 64  */
 65 Ptolemy.Quaternion.prototype.clone = function() {
 66     return new Ptolemy.Quaternion(this.x, this.y, this.z, this.w);

 77 /**
 78  * Computes the length of this Quaternion.
 79  * @return A Number with the quaternion's length.
 80  */
 81 Ptolemy.Quaternion.prototype.length = function() {
 82     return Math.sqrt(this.lengthSquared());
 83 };
 84 
 85 /**
 86  * Computes the squared length of this Quaternion.
 87  * @return A Number with the quaternion's squaredlength.
 88  */
 89 Ptolemy.Quaternion.prototype.lengthSquared = function() {
 90     return (this.x * this.x) + (this.y * this.y) + (this.z * this.z) + (this.w * this.w);
 91 };

 93 /**
 94  * Creates a new Quaternion instance wich corresponds to the normalization of
 95  * this one. This function check if the lenght of the quaternion is zero and
 96  * if so then return null.
 97  * @return A new Quaternion instance or null if the quaternion's lenght is zero.
 98  */
 99 Ptolemy.Quaternion.prototype.normalize = function() {
100 	var length = this.length();
101     if (length == 0) {
102         return null;
103     } else {
104         return new Ptolemy.Quaternion(
105         this.x / length,
106         this.y / length,
107         this.z / length,
108         this.w / length);
109     }
110 };

112 /**
113  * Creats a new instance which corresponds to the multiplication of this quaternion
114  * with the specified one, in the order: RESULT = THIS x THAT
115  */
116 Ptolemy.Quaternion.prototype.multiply = function(quaternion) {
117     return new Ptolemy.Quaternion(
118     (this.w * quaternion.x) + (this.x * quaternion.w) + (this.y * quaternion.z) - (this.z * quaternion.y),
119     (this.w * quaternion.y) + (this.y * quaternion.w) + (this.z * quaternion.x) - (this.x * quaternion.z),
120     (this.w * quaternion.z) + (this.z * quaternion.w) + (this.x * quaternion.y) - (this.y * quaternion.x),
121     (this.w * quaternion.w) - (this.x * quaternion.x) - (this.y * quaternion.y) - (this.z * quaternion.z)
122     );
123 };

125 /**
126  * Gets the roll angle represented by this quaternion.
127  * @returns Angle instance.
128  */
129 Ptolemy.Quaternion.prototype.getRoll = function() {
130     var radians = Math.atan2(
131     (2.0 * this.x * this.w) - (2.0 * this.y * this.z),
132     1.0 - 2.0 * (this.x * this.x) - 2.0 * (this.z * this.z)
133     );
134     return Ptolemy.Angle.fromRadians(radians);
135 };
136 
137 /**
138  * Gets the pitch angle represented by this quaternion.
139  * @returns Angle instance.
140  */
141 Ptolemy.Quaternion.prototype.getPitch = function() {
142     var radians = Math.atan2(
143     (2.0 * this.y * this.w) - (2.0 * this.x * this.z),
144     1.0 - (2.0 * this.y * this.y) - (2.0 * this.z * this.z)
145     );
146     return Ptolemy.Angle.fromRadians(radians);
147 };
148 
149 /**
150  * Gets the pityawch angle represented by this quaternion.
151  * @returns Angle instance.
152  */
153 Ptolemy.Quaternion.prototype.getYaw = function() {
154     var radians = Math.asin((2.0 * this.x * this.y) + (2.0 * this.z * this.w));
155     return Ptolemy.Angle.fromRadians(radians);
156 };

166 /**
167  * Creates a new quaternion instance from the specified matrix.
168  * See http://www.euclideanspace.com/maths/geometry/rotations/conversions/matrixToQuaternion/index.htm
169  * for details.

186         w = s * 0.25;
187     } else if ((matrix.m11 > matrix.m22) && (matrix.m11 > matrix.m33)) {
188         s = 2.0 * Math.sqrt(1.0 + matrix.m11 - matrix.m22 - matrix.m33);
189         x = s * 0.25;
190         y = (matrix.m21 + matrix.m12) / s;
191         z = (matrix.m13 + matrix.m31) / s;
192         w = (matrix.m32 - matrix.m23) / s;
193     } else if (matrix.m22 > matrix.m33) {
194         s = 2.0 * Math.sqrt(1.0 + matrix.m22 - matrix.m11 - matrix.m33);
195         x = (matrix.m21 + matrix.m12) / s;
196         y = s / 0.25;
197         z = (matrix.m32 + matrix.m23) / s;
198         w = (matrix.m13 - matrix.m31) / s;
199     } else {
200         s = 2.0 * Math.sqrt(1.0 + matrix.m33 - matrix.m11 - matrix.m22);
201         x = (matrix.m13 + matrix.m31) / s;
202         y = (matrix.m32 + matrix.m23) / s;
203         z = s / 0.25;
204         w = (matrix.m21 - matrix.m12) / s;
205     }
206     return new Ptolemy.Quaternion(x, y, z, w);
207 };

209 /**
210  * Creates a new quaternion instance from specified roll, pitch, yaw. The angles are applied in 
211  * the order roll, pitch, yaw.
212  * See: 
213  * http://en.wikipedia.org/wiki/Conversion_between_quaternions_and_Euler_angles
214  * http://upload.wikimedia.org/math/8/f/2/8f24d7035f36dd398e6253a521d7ac0c.png
215  * http://en.wikipedia.org/wiki/Tait-Bryan_angles
216  *
217  * @param {Angle} roll  Angle rotation about X axis.
218  * @param {Angle} pitch Angle rotation about Y axis.
219  * @param {Angle} yaw   Angle rotation about Z axis.
220  * @return Quaternion with the representation of the combined Roll-Pitch-Yaw rotation (in that order).
221  */
222 Ptolemy.Quaternion.fromRollPitchYaw = function(roll, pitch, yaw) {
223 
224     // Compute cos(angle/2) and sin(angle/2) for simplicifty.
225     var cosroll = roll.cosHalfAngle();
226     var cospitch = pitch.cosHalfAngle();
227     var cosyaw = yaw.cosHalfAngle();
228     var sinroll = roll.sinHalfAngle();
229     var sinpitch = pitch.sinHalfAngle();
230     var sinyaw = yaw.sinHalfAngle();
231 
232     var qw = (cosroll * cospitch * cosyaw) + (sinroll * sinpitch * sinyaw);
233     var qx = (sinroll * cospitch * cosyaw) - (cosroll * sinpitch * sinyaw);
234     var qy = (cosroll * sinpitch * cosyaw) + (sinroll * cospitch * sinyaw);
235     var qz = (cosroll * cospitch * sinyaw) - (sinroll * sinpitch * cosyaw);
236 
237     return new Ptolemy.Quaternion(qx, qy, qz, qw);
238 };
