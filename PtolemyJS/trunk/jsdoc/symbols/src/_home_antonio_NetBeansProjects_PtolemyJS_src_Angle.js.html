 20 /**
 21  * @fileoverview This contains class and functions related to Angles.
 22  * Operations with angles are common so it has sense to create a class with
 23  * the basic operations.
 24  *
 25  * @author Antonio Santiago asantiagop(at)gmail(dot)com
 26  */
 27 
 28 /**
 29  * Creates a new instance of Angle given a pair degree-radians values,
 30  * <b>Don't use this constructor directly. Instead use factory methods:</b>
 31  * @example
 32  * var angle = Ptolemy.Angle.fromDegrees(45);
 33  *
 34  * An Angle instance has two attributes: degrees and radians both storing the same
 35  * angle in different representation.
 36  *
 37  * @constructor
 38  */
 39 Ptolemy.Angle = function(degrees, radians) {
 40     this.degrees = degrees;
 41     this.radians = radians;
 42 };
 43 
 44 /**
 45  * Creates a new Angle instance as addition of this angle with the specified one, that is,
 46  * result = this + angle.
 47  * @param {Angle} angle 
 48  * @return New Angle instance
 49  */
 50 Ptolemy.Angle.prototype.add = function(angle) {
 51     return Ptolemy.Angle.fromDegrees(this.degrees + angle.degrees);
 52 };
 53 
 54 /**
 55  * Creates a new Angle instance as the substraction of the specified angle to this one, that is,
 56  * result = this-angle.
 57  * @param {Angle} angle 
 58  * @return New Angle instance
 59  */
 60 Ptolemy.Angle.prototype.substract = function(angle) {
 61     return Ptolemy.Angle.fromDegrees(this.degrees - angle.degrees);
 62 };
 63 
 64 /**
 65  * Computes the cosine of this angel.
 66  */
 67 Ptolemy.Angle.prototype.cos = function() {
 68     return Math.cos(this.radians);
 69 };
 70 
 71 /**
 72  * Computes the cosine of half this angel.
 73  */
 74 Ptolemy.Angle.prototype.cosHalfAngle = function() {
 75     return Math.cos(0.5 * this.radians);
 76 };
 77 
 78 /**
 79  * Computes the sine of this angel.
 80  */
 81 Ptolemy.Angle.prototype.sin = function() {
 82     return Math.sin(this.radians);
 83 };
 84 
 85 /**
 86  * Computes the sine of half this angel.
 87  */
 88 Ptolemy.Angle.prototype.sinHalfAngle = function() {
 89     return Math.sin(0.5 * this.radians);
 90 };
 91 
 92 /**
 93  * Angle constants and class methods.
 94  */
 95 Ptolemy.Angle.DEGREES_TO_RADIANS = Math.PI / 180.0;
 96 Ptolemy.Angle.RADIANS_TO_DEGREES = 180.0 / Math.PI;
 97 
 98 /**
 99  * Creates a new Angle instance from the specified degrees.
100  * @param {Number} degrees
101  */
102 Ptolemy.Angle.fromDegrees = function(degrees) {
103     return new Ptolemy.Angle(degrees, Ptolemy.Angle.DEGREES_TO_RADIANS * degrees);
104 };
105 
106 /**
107  * Creates a new Angle instance from the specified radians.
108  * @param {Number} radians
109  */
110 Ptolemy.Angle.fromRadians = function(radians) {
111     return new Ptolemy.Angle(radians * Ptolemy.Angle.RADIANS_TO_DEGREES, radians);
112 };
