 21  * @fileoverview This contains class and functions related to Vectors.
 22  * Althought Vectors have four dimensions operations works with the first
 23  * three component x,y,z. Operations that works for the four components
 24  * of the vector are named with a final '4', like 'add4' or 'substract4'.

 26  * @author Antonio Santiago asantiagop(at)gmail(dot)com

 29 /**
 30  * Creates a new Vector4 instance given the float values. The parameter 'w' is optional
 31  * and if it doesn't exists the DEFAULT_W constant is used.
 32  * @constructor
 33  * @param {Number} x
 34  * @param {Number} y
 35  * @param {Number} z
 36  * @param {Number} w
 37  */
 38 Ptolemy.Vector4 = function(x, y, z, w) {
 39     this.x = x;
 40     this.y = y;
 41     this.z = z;
 42     if (w==null) {
 43         this.w = Ptolemy.Vector4.DEFAULT_W();
 44     } else {
 45         this.w = w;
 46     }
 47 };
 48 
 49 /**
 50  * Check if this vector is equal to the specified vector.
 51  * @param {Vector4} vector Vector
 52  * @return True if they are equals, false otherwise.
 53  */
 54 Ptolemy.Vector4.prototype.equals = function(vector) {
 55     return (this.x == vector.x) && (this.y == vector.y) && (this.z == vector.z) && (this.w == vector.w);
 56 };

162 /**
163  * Computes the dot product (only for the x,y,z components) between this vector 
164  * and the specified one: this (dot) vector.
165  * @param {Vector} vector
166  * @return Number which corresponds to the dot product.
167  */
168 Ptolemy.Vector4.prototype.dot = function(vector) {
169     return (this.x * vector.x) + (this.y * vector.y) + (this.z * vector.z);
170 };
171 
172 /**
173  * Computes the dot product between this vector and the specified one: this (dot) vector,
174  * the dot is applied to the four components of the vector.
175  * @param {Vector} vector
176  * @return Number which corresponds to the dot product.
177  */
178 Ptolemy.Vector4.prototype.dot4 = function(vector) {
179     return (this.x * vector.x) + (this.y * vector.y) + (this.z * vector.z) + (this.w * vector.w);
180 };
181 
182 /**
183  * Computes the cross product (for x,y,z components) between this vector and the specified one.
184  * @param {Vector} vector.
185  * @returns New Vector instance.
186  */
187 Ptolemy.Vector4.prototype.cross = function(vector) {
188     return new Ptolemy.Vector4(
189     (this.y * vector.z) - (this.z * vector.y),
190     (this.z * vector.x) - (this.x * vector.z),
191     (this.x * vector.y) - (this.y * vector.x),
192     this.w
193     );
194 };

196 /**
197  * Computes the length of this Vector for the first three components x,y,z.
198  * @return a Number with the vector's length.
199  */
200 Ptolemy.Vector4.prototype.length = function() {
201     return Math.sqrt(this.lengthSquared());
202 };
203 
204 /**
205  * Computes the squared length of this Vector for the first three components x,y,z.
206  * @return a Number with the vector's length.
207  */
208 Ptolemy.Vector4.prototype.lengthSquared = function() {
209     return (this.x * this.x) + (this.y * this.y) + (this.z * this.z);
210 };

212 /**
213  * Creates a new Vector instance with corresponds with the normalization of this vector.
214  * This function check if the computed vector's lenght is zero and if so returns null.
215  * @returns New Vector instance or 'null' if the length of the vector is zero.
216  */
217 Ptolemy.Vector4.prototype.normalize = function() {
218     var length = this.length();
219     if (length == 0) {
220         return null;
221     } else {
222         return new Ptolemy.Vector4(
223         this.x / length,
224         this.y / length,
225         this.z / length,
226         this.w
227         );
228     }
229 };

231 /**
232  * Computes the squared distance to the specified Vector
233  * @param {Vector4} vector
234  * @return Squared distance.
235  */
236 Ptolemy.Vector4.prototype.distanceToSquared = function(vector) {
237     var distSquared = 0;
238     var tempOp = this.x - vector.x;
239     distSquared += tempOp * tempOp;
240     tempOp = this.y - vector.y;
241     distSquared += tempOp * tempOp;
242     tempOp = this.z - vector.z;
243     distSquared += tempOp * tempOp;
244     return distSquared;
245 };
246 
247 /**
248  * Computes the distance to the specified Vector
249  * @param {Vector4} vector
250  * @return Computed distance.
251  */
252 Ptolemy.Vector4.prototype.distance = function(vector) {
253     return Math.sqrt(this.distanceToSquared(vector));
254 };
