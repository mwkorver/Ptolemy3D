 20 /**
 21 * @fileoverview Globe contains the objects to be rendered.
 22 * The Globe can have one or more elevation models which provides the terrain elevations. 
 23 * Elevation models are used from the beggining to the end of the array to server terrain elevation 
 24 * information.
 25 * In the same way a Globe contains an array of RenderableObject which are rendered from the
 26 * beggining to the end of the array.
 27 * 
 28 * @author Antonio Santiago asantiagop(at)gmail(dot)com
 29 */
 30 
 31 /**
 32  * Creates a new Globe instance.
 33  * @constructor
 34  */
 35 Ptolemy.Globe = function() {
 36     this.sectorsArray = [];
 37     this.elevationModelsArray = [];
 38     this.renderableObjectsArray = [];
 39 
 40 	this.initializeSectors();
 41 };
 42 
 43 /**
 44 * Add a new elevation to the elevation model array. 
 45 * @param elevationModel
 46 */
 47 Ptolemy.Globe.prototype.addElevationModel = function(elevationModel) {
 48     this.elevationModelArray.push(elevationModel);
 49 };
 50 
 51 /**
 52 * Adds a new renderable object to renderable objects array.
 53 * @param renderableObject
 54 */
 55 Ptolemy.Globe.prototype.addRenderableObject = function(renderableObject) {
 56     this.renderableObjectsArray.push(renderableObject);
 57 };
 58 
 59 Ptolemy.Globe.prototype.initializeSectors = function() {
 60     for (var lat = Ptolemy.Globe.MIN_LAT; lat < Ptolemy.Globe.MAX_LAT; lat += Ptolemy.Globe.LAT_STEP) {
 61         for (var lon = Ptolemy.Globe.MIN_LON; lon < Ptolemy.Globe.MAX_LON; lon += Ptolemy.Globe.LON_STEP) {
 62             var sector = new Ptolemy.Sector(lat, lon, lat + Ptolemy.Globe.LAT_STEP, lon + Ptolemy.Globe.LON_STEP);
 63             this.addRenderableObject(sector);
 64         }
 65     }
 66 };

 72 // TODO - Globe is considered a sphere so translation methods translates from cartesian
 73 //		to spherical coordinates.
 74 //		We will need to improve this and use some elipsoid to represent earth and more
 75 //		acurate values, mainly when computing distances, etc.
 76 Ptolemy.Globe.RADIUS = function() {
 77     return 6378137.0;
 78 };

 80 /**
 81  * Computes the geographic position from given cartesion coordinates.
 82  *
 83  * NOTE: This function cosiders the earth as an sphere. See
 84  * See: http://en.wikipedia.org/wiki/Spherical_coordinate_system
 85  *      http://mathworld.wolfram.com/SphericalCoordinates.html
 86  *
 87  * @param {Vector4} vector The cartesion position to convert to geographic.
 88  * @return A Position instance containing the geographic position.
 89  */
 90 Ptolemy.Globe.computeGeographicFromCartesion = function(vector) {
 91     var radius = vector.length();
 92     var theta = Math.atan2(vector.y / vector.x);
 93     var phi = Mat.acos(vector.x / radius);
 94 
 95     var lat = Ptolemy.Angle.fromDegrees(90).substract(phi);
 96     var lon = Ptolemy.Angle.fromDegrees(180).substract(theta);
 97     var latlon = new Ptolemy.LatLon(lat, lon);
 98 
 99     return new Ptolemy.Position(latlon, radius);
100 };
101 
102 /**
103  * Computes cartesion position (x,y,z) in model coordinates from given
104  * geographic position.
105  *
106  * NOTE: This function cosiders the earth as an sphere. See
107  * See: http://en.wikipedia.org/wiki/Spherical_coordinate_system
108  *      http://mathworld.wolfram.com/SphericalCoordinates.html
109  *
110  * @param {Position} position The geographic position to convert to cartesion.
111  * @return A Vector4 instance with x,y,z componentes containing the cartesion values.
112  */
113 Ptolemy.Globe.computeCartesionFromGeographic = function(position) {
114     var lat = position.latlon.latitude;
115     var lon = position.latlon.longitude;
116     var elev = position.elevation;
117 
118     var phi = Ptolemy.Angle.fromDegrees(90).substract(lat);
119     var theta = Ptolemy.Angle.fromDegrees(180).add(lon);
120     var radius = Ptolemy.Globe.RADIUS() + elev;
121 
122     var x = radius * theta.cos() * phi.sin();
123     var y = radius * theta.sin() * phi.sin();
124     var z = radius * phi.cos();
125 
126     return new Ptolemy.Vector4(x, y, z);
127 };

130 Ptolemy.Globe.LAT_STEP = 10;
131 Ptolemy.Globe.LON_STEP = 10;

133 Ptolemy.Globe.MIN_LAT = -90;

140 /***************************************************************
141 * Sector
142 */
143 Ptolemy.Sector = function(minLat, minLon, maxLat, maxLon) {
144     this.minLat = minLat;
145     this.minLon = minLon;
146     this.maxLat = maxLat;
147     this.maxLon = maxLon;
148 
149     this.initialized = false;
150     this.sectorVertices = null;
151     this.sectorVerticesBuffer = null;
152     this.sectorIndices = null;
153     this.sectorIndicesBuffer = null;
154 };
155 
156 Ptolemy.Sector.prototype.initialize = function(dc) {
157     var gl = dc.gl;
158     var p1 = new Ptolemy.Position(new Ptolemy.LatLon(
159 		Ptolemy.Angle.fromDegrees(this.minLat), 
160 		Ptolemy.Angle.fromDegrees(this.minLon)), 
161 		Ptolemy.Globe.RADIUS());
162     var p2 = new Ptolemy.Position(new Ptolemy.LatLon(
163 		Ptolemy.Angle.fromDegrees(this.minLat), 
164 		Ptolemy.Angle.fromDegrees(this.maxLon)), 
165 		Ptolemy.Globe.RADIUS());
166     var p3 = new Ptolemy.Position(new Ptolemy.LatLon(
167 		Ptolemy.Angle.fromDegrees(this.maxLat), 
168 		Ptolemy.Angle.fromDegrees(this.maxLon)), 
169 		Ptolemy.Globe.RADIUS());
170     var p4 = new Ptolemy.Position(new Ptolemy.LatLon(
171 		Ptolemy.Angle.fromDegrees(this.maxLat), 
172 		Ptolemy.Angle.fromDegrees(this.minLon)), 
173 		Ptolemy.Globe.RADIUS());
174 
175 	var v1 = Ptolemy.Globe.computeCartesionFromGeographic(p1);
176 	var v2 = Ptolemy.Globe.computeCartesionFromGeographic(p2);
177 	var v3 = Ptolemy.Globe.computeCartesionFromGeographic(p3);
178 	var v4 = Ptolemy.Globe.computeCartesionFromGeographic(p4);
179 
180     this.sectorVertices = [
181     v1.x, v1.y, v1.z,
182     v2.x, v2.y, v2.z,
183     v3.x, v3.y, v3.z,
184     v4.x, v4.y, v4.z
185     ];
186 
187     this.sectorIndices = [0, 1, 2, 3];
188 
189     this.sectorVerticesBuffer = gl.createBuffer();
190     gl.bindBuffer(gl.ARRAY_BUFFER, this.sectorVerticesBuffer);
191     gl.bufferData(gl.ARRAY_BUFFER, new WebGLFloatArray(this.sectorVertices), gl.STATIC_DRAW);
192     this.sectorVerticesBuffer.itemSize = 3;
193     this.sectorVerticesBuffer.numItems = 4;
194 
195     this.sectorIndicesBuffer = gl.createBuffer();
196     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.sectorIndicesBuffer);
197     gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new WebGLUnsignedShortArray(this.sectorIndices), gl.STATIC_DRAW);
198     this.sectorIndicesBuffer.itemSize = 1;
199     this.sectorIndicesBuffer.numItems = 4;
200 
201     this.initialized = true;
202 };
203 
204 Ptolemy.Sector.prototype.render = function(dc) {
205     var gl = dc.gl;
206     if (!this.initialized) {
207         this.initialize(dc);
208     }
209 
210     gl.bindBuffer(gl.ARRAY_BUFFER, this.sectorVerticesBuffer);
211     gl.vertexAttribPointer(Ptolemy.shaderProgram.vertexPositionAttribute, this.sectorVerticesBuffer.itemSize, gl.FLOAT, false, 0, 0);
212 
213     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.sectorIndicesBuffer);
214     gl.drawElements(gl.LINE_LOOP, this.sectorIndicesBuffer.numItems, gl.UNSIGNED_SHORT, 0);
215 };
