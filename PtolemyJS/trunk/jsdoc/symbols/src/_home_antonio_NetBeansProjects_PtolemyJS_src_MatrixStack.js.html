 20 /**
 21  * @fileoverview MatrixStack tries to emulate the matrix stack offered by OpenGL.
 22  * MatrixStack allows to push and pop matrices (implemented as Matrix objects).
 23  *
 24  * @author Antonio Santiago asantiagop(at)gmail(dot)com
 25  */
 26 
 27 /**
 28  * Creates a new MatrixStack instance.
 29  */
 30 Ptolemy.MatrixStack = function() {
 31     this.matrixStackArray = [];
 32 };
 33 
 34 /**
 35  * Pushes the current matrix stack by one, duplicating the current matrix
 36  */
 37 Ptolemy.MatrixStack.prototype.push = function() {
 38     var current = this.current();
 39     if (current) {
 40         this.matrixStackArray.push(current.clone());
 41     }
 42 };
 43 
 44 /**
 45  * Pops the current matrix stack, replacing the current matrix with the one below it on the stack
 46  * @returns the removed matrix
 47  */
 48 Ptolemy.MatrixStack.prototype.pop = function() {
 49     if (this.matrixStackArray.length > 0) {
 50         return this.matrixStackArray.pop();
 51     } else {
 52         return null;
 53     }
 54 };
 55 
 56 /**
 57  * Clears the stack.
 58  */
 59 Ptolemy.MatrixStack.prototype.clear = function() {
 60     this.matrixStackArray = [];
 61 };
 62 
 63 /**
 64  * Replace the current matrix with the specified matrix
 65  * @param {Matrix} matrix
 66  */
 67 Ptolemy.MatrixStack.prototype.load = function(matrix) {
 68     this.matrixStackArray.pop();
 69     this.matrixStackArray.push(matrix);
 70 };
 71 
 72 /**
 73  * Gets the matrix at top of the stack or null if it is empty.
 74  */
 75 Ptolemy.MatrixStack.prototype.current = function() {
 76     var length = this.matrixStackArray.length;
 77     if (length > 0) {
 78         return this.matrixStackArray[length - 1];
 79     }
 80     return null;
 81 };
 82 
 83 /**
 84  * Multiply the current matrix with the specified matrix
 85  * @param {Matrix} matrix
 86  */
 87 Ptolemy.MatrixStack.prototype.mult = function(matrix) {
 88     var current = this.current();
 89     if (current) {
 90         var newCurrent = current.multiply(matrix);
 91         this.load(newCurrent);
 92     } else {
 93         this.load(matrix);
 94     }
 95 };
