 20 /**
 21  * @fileoverview Ptolemy WebGL Virtual Globe 2010

 28 /**
 29  * @function This function is executed when Ptolemy.js code is imported in 
 30  * your page and is responsible to define the Ptolemy namespace with some common
 31  * methods.
 32  */
 33 (function() {
 34     var loadedScripts = {};
 35 
 36     /**
 37      * @namespace Ptolemy namespace contains all classes, methods and properties
 38      * related to Ptolemy.
 39      */
 40     Ptolemy = {
 41         scriptName: 'Ptolemy.js',
 42         version: '0.0.1',
 43         versionName: 'nano',
 44 
 45         // TODO - Not used, must be implemented in the right way
 46         /**
 47          * Computes the base path of 'Ptolemy.scriptName'.
 48 	 *
 49 	 * @return path
 50 	 */
 51         getBasePath: function() {
 52             var scripts = document.getElementsByTagName("script");
 53             var scriptSrc = null;
 54             for (var i = 0; i < scripts.length; i++) {
 55                 if (scripts[i].src && scripts[i].src.indexOf(Ptolemy.scriptName) != -1) {
 56                     scriptSrc = scripts[i].src;
 57                     break;
 58                 }
 59             }
 60             if (scriptSrc == null) {
 61                 alert("Can't get the base path for script name: '" + Ptolemy.scriptName + "'.");
 62                 return;
 63             }
 64             var parts = scriptSrc.split("/");
 65             parts.pop();
 66             var basePath = parts.join("/");
 67 
 68             return basePath;
 69         },
 70 
 71         // TODO - Implement function in the right way and test in some borwsers.
 72         /**
 73 	 * Private function responsible to include only once required JavaScript files. 
 74 	 * The function computes the base location to the Ptolemy.js script and tries
 75 	 * to load all scripts from that base path.
 76 	 * 
 77 	 * @param jsFile Path to the file. Must be relative to Ptolemy.js script location.
 78 	 */        
 79         includeFile: function(jsFile) {
 80             if (loadedScripts[jsFile] != null) return;
 81 
 82             var basePath = Ptolemy.getBasePath();
 83 
 84             var head = document.getElementsByTagName("head")[0];
 85             var scriptElement = document.createElement('script');
 86             scriptElement.setAttribute('language', 'javascript');
 87             scriptElement.setAttribute('type', 'text/javascript');
 88             scriptElement.setAttribute('src', jsFile);
 89             head.appendChild(scriptElement);
 90             loadedScripts[jsFile] = jsFile;
 91         },
 92 
 93         shaderProgram: null,
 94         // TODO - Improve shaders manipulation.
 95         /**
 96 	 * Initialize the shaders.
 97 	 * @param gl GL context reference.
 98 	 * @return A shader program with vertex and fragment shader attached.
 99 	 */        
100         initializeShaders: function(gl) {
101             var fragmentShader = Ptolemy._getShader(gl, "shader-fs");
102             var vertexShader = Ptolemy._getShader(gl, "shader-vs");
103 
104             this.shaderProgram = gl.createProgram();
105             gl.attachShader(this.shaderProgram, vertexShader);
106             gl.attachShader(this.shaderProgram, fragmentShader);
107             gl.linkProgram(this.shaderProgram);
108 
109             if (!gl.getProgramParameter(this.shaderProgram, gl.LINK_STATUS)) {
110                 alert("Could not initialise shaders");
111             }
112 
113             gl.useProgram(this.shaderProgram);
114 
115             this.shaderProgram.vertexPositionAttribute = gl.getAttribLocation(this.shaderProgram, "aVertexPosition");
116             gl.enableVertexAttribArray(this.shaderProgram.vertexPositionAttribute);
117 
118             this.shaderProgram.pMatrixUniform = gl.getUniformLocation(this.shaderProgram, "uPMatrix");
119             this.shaderProgram.mvMatrixUniform = gl.getUniformLocation(this.shaderProgram, "uMVMatrix");
120         },
121 
122         /**
123 	 * Parse the specified shader and compile the program.
124 	 * @private
125 	 * @param gl
126 	 * @param id
127 	 */
128         _getShader: function(gl, id) {
129             var shaderScript = document.getElementById(id);
130             if (!shaderScript) {
131                 return null;
132             }
133 
134             var str = "";
135             var k = shaderScript.firstChild;
136             while (k) {
137                 if (k.nodeType == 3) {
138                     str += k.textContent;
139                 }
140                 k = k.nextSibling;
141             }
142 
143             var shader;
144             if (shaderScript.type == "x-shader/x-fragment") {
145                 shader = gl.createShader(gl.FRAGMENT_SHADER);
146             } else if (shaderScript.type == "x-shader/x-vertex") {
147                 shader = gl.createShader(gl.VERTEX_SHADER);
148             } else {
149                 return null;
150             }
151 
152             gl.shaderSource(shader, str);
153             gl.compileShader(shader);
154 
155             if (!gl.getShaderParameter(shader, gl.COMPILE_STATUS)) {
156                 alert(gl.getShaderInfoLog(shader));
157                 return null;
158             }
159 
160             return shader;
161         }
162     };
